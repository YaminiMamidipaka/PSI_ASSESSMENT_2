object Coupon
{
	def newCouponCodes(Array:Array[String]):Unit=
	{
		for(i <- Array){
			println(i + " - " + i.length);
		}
	}
	
	def main(arg:Array[String])
	{	
		var couponCode = Array("A","BB","CCC","DDDD","EEEEE")
		newCouponCodes(couponCode);
	}
}
